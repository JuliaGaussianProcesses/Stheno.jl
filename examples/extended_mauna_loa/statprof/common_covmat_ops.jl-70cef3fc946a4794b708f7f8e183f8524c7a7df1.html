37 function update_chol(chol::Cholesky, C12::AbstractMatrix, C22::AbstractMatrix)
38 U12 = chol.U' \ C12
39 U22 = cholesky(Symmetric(C22 - U12'U12)).U
40 return Cholesky([chol.U U12; zero(U12') U22], 'U', 0)
41 end
