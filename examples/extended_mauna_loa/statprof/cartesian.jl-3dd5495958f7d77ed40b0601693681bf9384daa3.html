13

15 `rangeexpr` may be an anonymous-function expression, or a simple symbol `var` in which case
16 the range is `axes(var, d)` for dimension `d`.
17

20

33 end

89 """
90 @ncall N f sym...
91
92 Generate a function call expression. `sym` represents any number of function arguments, the
93 last of which may be an anonymous-function expression and is expanded into `N` arguments.
94
95 For example, `@ncall 3 func a` generates
96

161 @nall N expr
162
163 Check whether all of the expressions generated by the anonymous-function expression `expr`
164 evaluate to `true`.
165

204 """
205 @nif N conditionexpr expr
206 @nif N conditionexpr expr elseexpr
207
208 Generates a sequence of `if ... elseif ... else ... end` statements. For example:
209
210 @nif 3 d->(i_d >= size(A,d)) d->(error("Dimension ", d, " too big")) d->println("All OK")
211
212 would generate:
213
214 if i_1 > size(A, 1)
215 error("Dimension ", 1, " too big")
216 elseif i_2 > size(A, 2)
217 error("Dimension ", 2, " too big")
218 else
219 println("All OK")
220 end
221 """
222 macro nif(N, condition, operation...)
223 # Handle the final "else"
224 ex = esc(inlineanonymous(length(operation) > 1 ? operation[2] : operation[1], N))
225 # Make the nested if statements
226 for i = N-1:-1:1
227 ex = Expr(:if, esc(inlineanonymous(condition,i)), esc(inlineanonymous(operation[1],i)), ex)
228 end

234 # Simplify expressions like :(d->3:size(A,d)-3) given an explicit value for d
235 function inlineanonymous(ex::Expr, val)
236 if ex.head !== :->
237 throw(ArgumentError("not an anonymous function"))
238 end
239 if !isa(ex.args[1], Symbol)
240 throw(ArgumentError("not a single-argument anonymous function"))
241 end
242 sym = ex.args[1]::Symbol
243 ex = ex.args[2]::Expr
244 exout = lreplace(ex, sym, val)
245 exout = poplinenum(exout)
246 exprresolve(exout)
247 end

252 # Replace a symbol by a value or a "coded" symbol
253 # E.g., for d = 3,
254 # lreplace(:d, :d, 3) -> 3
255 # lreplace(:i_d, :d, 3) -> :i_3
256 # lreplace(:i_{d-1}, :d, 3) -> :i_2
257 # This follows LaTeX notation.
258 struct LReplace{S<:AbstractString}
259 pat_sym::Symbol
260 pat_str::S
261 val::Int
262 end
263 LReplace(sym::Symbol, val::Integer) = LReplace(sym, string(sym), val)

272 function lreplace!(str::AbstractString, r::LReplace)
273 i = firstindex(str)
274 pat = r.pat_str
275 j = firstindex(pat)
276 matching = false
277 local istart::Int
278 while i <= ncodeunits(str)
279 cstr = str[i]
280 i = nextind(str, i)
281 if !matching
282 if cstr != '_' || i > ncodeunits(str)
283 continue
284 end
285 istart = i
286 cstr = str[i]
287 i = nextind(str, i)
288 end
289 if j <= lastindex(pat)
290 cr = pat[j]
291 j = nextind(pat, j)
292 if cstr == cr
293 matching = true
294 else
295 matching = false
296 j = firstindex(pat)
297 i = istart
298 continue
299 end
300 end
301 if matching && j > lastindex(pat)
302 if i > lastindex(str) || str[i] == '_'
303 # We have a match
304 return string(str[1:prevind(str, istart)], r.val, lreplace!(str[i:end], r))
305 end
306 matching = false
307 j = firstindex(pat)
308 i = istart
309 end
310 end
311 str
312 end

314 function lreplace!(ex::Expr, r::LReplace)
315 # Curly-brace notation, which acts like parentheses
316 if ex.head === :curly && length(ex.args) == 2 && isa(ex.args[1], Symbol) && endswith(string(ex.args[1]::Symbol), "_")
317 excurly = exprresolve(lreplace!(ex.args[2], r))
318 if isa(excurly, Int)
319 return Symbol(ex.args[1]::Symbol, excurly)

322 return ex
323 end
324 end
325 for i in 1:length(ex.args)
326 ex.args[i] = lreplace!(ex.args[i], r)

350 const exprresolve_arith_dict = Dict{Symbol,Function}(:+ => +,
351 :- => -, :* => *, :/ => /, :^ => ^, :div => div)
352 const exprresolve_cond_dict = Dict{Symbol,Function}(:(==) => ==,
353 :(<) => <, :(>) => >, :(<=) => <=, :(>=) => >=)

355 function exprresolve_arith(ex::Expr)
356 if ex.head === :call
357 callee = ex.args[1]
358 if isa(callee, Symbol)
359 if haskey(exprresolve_arith_dict, callee) && all(Bool[isa(ex.args[i], Number) for i = 2:length(ex.args)])
360 return true, exprresolve_arith_dict[callee](ex.args[2:end]...)
361 end
362 end
363 end
364 false, 0
365 end

369 function exprresolve_conditional(ex::Expr)
370 if ex.head === :call
371 callee = ex.args[1]
372 if isa(callee, Symbol)
373 if callee ∈ keys(exprresolve_cond_dict) && isa(ex.args[2], Number) && isa(ex.args[3], Number)
374 return true, exprresolve_cond_dict[callee](ex.args[2], ex.args[3])
375 end
376 end
377 end
378 false, false
379 end

383 function exprresolve(ex::Expr)
384 for i = 1:length(ex.args)
385 ex.args[i] = exprresolve(ex.args[i])
386 end
387 # Handle simple arithmetic
388 can_eval, result = exprresolve_arith(ex)
389 if can_eval
390 return result
391 elseif ex.head === :call && (ex.args[1] === :+ || ex.args[1] === :-) && length(ex.args) == 3 && ex.args[3] == 0
392 # simplify x+0 and x-0
393 return ex.args[2]
394 end
395 # Resolve array references
396 if ex.head === :ref && isa(ex.args[1], Array)
397 for i = 2:length(ex.args)
398 if !isa(ex.args[i], Real)
399 return ex
400 end
401 end
402 return ex.args[1][ex.args[2:end]...]
403 end
404 # Resolve conditionals
405 if ex.head === :if
406 can_eval, tf = exprresolve_conditional(ex.args[1])
407 if can_eval
408 ex = tf ? ex.args[2] : ex.args[3]
409 end
410 end
411 ex
412 end
