297 """
298 bcf = flatten(bc)
299
300 Create a "flat" representation of a lazy-broadcast operation.
301 From
302 f.(a, g.(b, c), d)
303 we produce the equivalent of
304 h.(a, b, c, d)
305 where
306 h(w, x, y, z) = f(w, g(x, y), z)
307 In terms of its internal representation,
308 Broadcasted(f, a, Broadcasted(g, b, c), d)
309 becomes
310 Broadcasted(h, a, b, c, d)
311
312 This is an optional operation that may make custom implementation of broadcasting easier in
313 some cases.
314 """
315 function flatten(bc::Broadcasted{Style}) where {Style}
316 isflat(bc) && return bc
317 # concatenate the nested arguments into {a, b, c, d}
318 args = cat_nested(bc)
319 # build a function `makeargs` that takes a "flat" argument list and
320 # and creates the appropriate input arguments for `f`, e.g.,
321 # makeargs = (w, x, y, z) -> (w, g(x, y), z)
322 #
323 # `makeargs` is built recursively and looks a bit like this:
324 # makeargs(w, x, y, z) = (w, makeargs1(x, y, z)...)
325 # = (w, g(x, y), makeargs2(z)...)
326 # = (w, g(x, y), z)
327 let makeargs = make_makeargs(()->(), bc.args), f = bc.f
328 newf = @inline function(args::Vararg{Any,N}) where N
329 f(makeargs(args...)...)
330 end
331 return Broadcasted{Style}(newf, args, bc.axes)
332 end
333 end

1049 function copyto_nonleaf!(dest, bc::Broadcasted, iter, state, count)

1052 y = iterate(iter, state)
1053 y === nothing && break

1056 if val isa T || typeof(val) === T
1057 @inbounds dest[I] = val

1063 end

1225 Expr(:., dotargs[1], Expr(:tuple, dotargs[2:end]...))

1232 Expr(:let, undot(dotargs[1]), dotargs[2])
1233 elseif x.head === :for # don't add dots to for x=... assignments
1234 Expr(:for, undot(dotargs[1]), dotargs[2])
1235 elseif (x.head === :(=) || x.head === :function || x.head === :macro) &&
1236 Meta.isexpr(x.args[1], :call) # function or macro definition
1237 Expr(x.head, x.args[1], dotargs[2])
