63 # the inverse of typeof_tfunc
64 # returns (type, isexact, isconcrete, istype)
65 # if isexact is false, the actual runtime type may (will) be a subtype of t
66 # if isconcrete is true, the actual runtime type is definitely concrete (unreachable if not valid as a typeof)
67 # if istype is true, the actual runtime value will definitely be a type (e.g. this is false for Union{Type{Int}, Int})
68 function instanceof_tfunc(@nospecialize(t))
69 if isa(t, Const)
70 if isa(t.val, Type)
71 return t.val, true, isconcretetype(t.val), true
72 end
73 return Bottom, true, false, false # runtime throws on non-Type
74 end
75 t = widenconst(t)
76 if t === Bottom
77 return Bottom, true, true, false # runtime unreachable
78 elseif t === typeof(Bottom) || typeintersect(t, Type) === Bottom
79 return Bottom, true, false, false # literal Bottom or non-Type
80 elseif isType(t)
81 tp = t.parameters[1]
82 return tp, !has_free_typevars(tp), isconcretetype(tp), true
83 elseif isa(t, UnionAll)
84 t′ = unwrap_unionall(t)
85 t′′, isexact, isconcrete, istype = instanceof_tfunc(t′)
86 tr = rewrap_unionall(t′′, t)
87 if t′′ isa DataType && t′′.name !== Tuple.name && !has_free_typevars(tr)
88 # a real instance must be within the declared bounds of the type,
89 # so we can intersect with the original wrapper.
90 tr = typeintersect(tr, t′′.name.wrapper)
91 isconcrete = !isabstracttype(t′′)
92 if tr === Union{}
93 # runtime unreachable (our inference Type{T} where S is
94 # uninhabited with any runtime T that exists)
95 isexact = true
96 end
97 end
98 return tr, isexact, isconcrete, istype
99 elseif isa(t, Union)
100 ta, isexact_a, isconcrete_a, istype_a = instanceof_tfunc(t.a)
101 tb, isexact_b, isconcrete_b, istype_b = instanceof_tfunc(t.b)
102 isconcrete = isconcrete_a && isconcrete_b
103 istype = istype_a && istype_b
104 # most users already handle the Union case, so here we assume that
105 # `isexact` only cares about the answers where there's actually a Type
106 # (and assuming other cases causing runtime errors)
107 ta === Union{} && return tb, isexact_b, isconcrete, istype
108 tb === Union{} && return ta, isexact_a, isconcrete, istype
109 return Union{ta, tb}, false, isconcrete, istype # at runtime, will be exactly one of these
110 end
111 return Any, false, false, false
112 end
