19 CartesianIndex(i, j, k...) -> I
20 CartesianIndex((i, j, k...)) -> I
21
22 Create a multidimensional index `I`, which can be used for
23 indexing a multidimensional array `A`. In particular, `A[I]` is
24 equivalent to `A[i,j,k...]`. One can freely mix integer and
25 `CartesianIndex` indices; for example, `A[Ipre, i, Ipost]` (where
26 `Ipre` and `Ipost` are `CartesianIndex` indices and `i` is an
27 `Int`) can be a useful expression when writing algorithms that
28 work along a single dimension of an array of arbitrary
29 dimensionality.
30
31 A `CartesianIndex` is sometimes produced by [`eachindex`](@ref), and
32 always when iterating with an explicit [`CartesianIndices`](@ref).

41

1116 @noinline function circshift!(dest::AbstractArray{T,N}, src, shiftamt::DimsInteger) where {T,N}
1117 dest === src && throw(ArgumentError("dest and src must be separate arrays"))
1118 inds = axes(src)
1119 axes(dest) == inds || throw(ArgumentError("indices of src and dest must match (got $inds and $(axes(dest)))"))

1126
1127 # For each dimension, we copy the first half of src to the second half
1128 # of dest, and the second half of src to the first half of dest. This
1129 # uses a recursive bifurcation strategy so that these splits can be
1130 # encoded by ranges, which means that we need only one call to `mod`
1131 # per dimension rather than one call per index.
1132 # `rdest` and `rsrc` are tuples-of-ranges that grow one dimension at a
1133 # time; when all the dimensions have been filled in, you call `copyto!`
1134 # for that block. In other words, in two dimensions schematically we
1135 # have the following call sequence (--> means a call):
1136 # circshift!(dest, src, shiftamt) -->
1137 # _circshift!(dest, src, ("first half of dim1",)) -->
1138 # _circshift!(dest, src, ("first half of dim1", "first half of dim2")) --> copyto!
1139 # _circshift!(dest, src, ("first half of dim1", "second half of dim2")) --> copyto!
1140 # _circshift!(dest, src, ("second half of dim1",)) -->
1141 # _circshift!(dest, src, ("second half of dim1", "first half of dim2")) --> copyto!
1142 # _circshift!(dest, src, ("second half of dim1", "second half of dim2")) --> copyto!
1143 @inline function _circshift!(dest, rdest, src, rsrc,
1144 inds::Tuple{AbstractUnitRange,Vararg{Any}},
1145 shiftamt::Tuple{Integer,Vararg{Any}})::typeof(dest)
1146 ind1, d = inds[1], shiftamt[1]
1147 s = mod(d, length(ind1))
1148 sf, sl = first(ind1)+s, last(ind1)-s
1149 r1, r2 = first(ind1):sf-1, sf:last(ind1)
1150 r3, r4 = first(ind1):sl, sl+1:last(ind1)
1151 tinds, tshiftamt = tail(inds), tail(shiftamt)
1152 _circshift!(dest, (rdest..., r1), src, (rsrc..., r4), tinds, tshiftamt)
1153 _circshift!(dest, (rdest..., r2), src, (rsrc..., r3), tinds, tshiftamt)
1154 end

1156 function _circshift!(dest, rdest, src, rsrc, inds, shiftamt)

1165 `src` and `dest` must have the same size, but can be offset in

1167 arrays have overlapping indices, then on the domain of the overlap
1168 `dest` agrees with `src`.

1185 8 12 16 4
1186 5 9 13 1
1187 6 10 14 2
1188 7 11 15 3
1189
1190 julia> dest[1:3,2:4] == src[1:3,2:4]
1191 true
1192 ```
1193 """
1194 function circcopy!(dest, src)
1195 dest === src && throw(ArgumentError("dest and src must be separate arrays"))

1197 if (szsrc = map(length, indssrc)) != (szdest = map(length, indsdest))
1198 throw(DimensionMismatch("src and dest must have the same sizes (got $szsrc and $szdest)"))
1199 end
1200 shift = map((isrc, idest)->first(isrc)-first(idest), indssrc, indsdest)
1201 all(x->x==0, shift) && return copyto!(dest, src)
1202 _circcopy!(dest, (), indsdest, src, (), indssrc)

1307 # convert(::Type{Bool}, x::Real) = x==0 ? false : x==1 ? true : throw(InexactError(...))
1308 # they're used to pre-emptively check in bulk when possible, which is much faster.
1309 # Also, the functions can be overloaded for custom types T<:Real :
1310 # a) in the unlikely eventuality that they use a different logic for Bool conversion
1311 # b) to skip the check if not necessary
1312 @inline try_bool_conversion(x::Real) =
1313 x == 0 || x == 1 || throw(InexactError(:try_bool_conversion, Bool, x))
1314 @inline unchecked_bool_convert(x::Real) = x == 1

1316 function copy_to_bitarray_chunks!(Bc::Vector{UInt64}, pos_d::Int, C::StridedArray{<:Real}, pos_s::Int, numbits::Int)
1317 @inbounds for i = (1:numbits) .+ (pos_s - 1)
1318 try_bool_conversion(C[i])
1319 end
1320
1321 kd0, ld0 = get_chunks_id(pos_d)
1322 kd1, ld1 = get_chunks_id(pos_d + numbits - 1)
1323
1324 delta_kd = kd1 - kd0
1325
1326 u = _msk64
1327 if delta_kd == 0
1328 msk_d0 = msk_d1 = ~(u << ld0) | (u << (ld1+1))
1329 lt0 = ld1
1330 else
1331 msk_d0 = ~(u << ld0)
1332 msk_d1 = (u << (ld1+1))
1333 lt0 = 63
1334 end
1335
1336 bind = kd0
1337 ind = pos_s
1338 @inbounds if ld0 > 0
1339 c = UInt64(0)
1340 for j = ld0:lt0
1341 c |= (UInt64(unchecked_bool_convert(C[ind])) << j)
1342 ind += 1
1343 end
1344 Bc[kd0] = (Bc[kd0] & msk_d0) | (c & ~msk_d0)
1345 bind += 1
1346 end
1347

1349 @inbounds for i = 1:nc
1350 c = UInt64(0)
1351 for j = 0:63
1352 c |= (UInt64(unchecked_bool_convert(C[ind])) << j)
1353 ind += 1
1354 end

1356 bind += 1
1357 end
1358
1359 @inbounds if bind ≤ kd1
1360 @assert bind == kd1
1361 c = UInt64(0)
1362 for j = 0:ld1
1363 c |= (UInt64(unchecked_bool_convert(C[ind])) << j)
1364 ind += 1
1365 end
