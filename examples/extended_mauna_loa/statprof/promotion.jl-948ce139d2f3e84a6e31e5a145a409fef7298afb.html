143 Compute a type that contains both `T` and `S`, which could be
144 either a parent of both types, or a `Union` if appropriate.

158 function promote_typejoin(@nospecialize(a), @nospecialize(b))
159 c = typejoin(_promote_typesubtract(a), _promote_typesubtract(b))

161 end

164 function promote_typejoin_union(::Type{T}) where T
165 if T === Union{}
166 return Union{}
167 elseif T isa UnionAll
168 return Any # TODO: compute more precise bounds
169 elseif T isa Union
170 return promote_typejoin(promote_typejoin_union(T.a), promote_typejoin_union(T.b))
171 elseif T isa DataType

174 else

179 function typejoin_union_tuple(T::DataType)
180 @_pure_meta
181 u = Base.unwrap_unionall(T)
182 p = (u::DataType).parameters
183 lr = length(p)::Int
184 if lr == 0
185 return Tuple{}
186 end
187 c = Vector{Any}(undef, lr)
188 for i = 1:lr
189 pi = p[i]
190 U = Core.Compiler.unwrapva(pi)
191 if U === Union{}
192 ci = Union{}
193 elseif U isa Union
194 ci = typejoin(U.a, U.b)
195 elseif U isa UnionAll
196 return Any # TODO: compute more precise bounds
197 else
198 ci = promote_typejoin_union(U)
199 end
200 if i == lr && Core.Compiler.isvarargtype(pi)
201 c[i] = isdefined(pi, :N) ? Vararg{ci, pi.N} : Vararg{ci}
202 else
203 c[i] = ci
204 end
205 end
206 return Base.rewrap_unionall(Tuple{c...}, T)

236 """
237 promote_type(type1, type2)
238
239 Promotion refers to converting values of mixed types to a single common type.
240 `promote_type` represents the default promotion behavior in Julia when
241 operators (usually mathematical) are given arguments of differing types.
242 `promote_type` generally tries to return a type which can at least approximate
243 most values of either input type without excessively widening. Some loss is
244 tolerated; for example, `promote_type(Int64, Float64)` returns
245 [`Float64`](@ref) even though strictly, not all [`Int64`](@ref) values can be
246 represented exactly as `Float64` values.
247
248 See also: [`promote`](@ref), [`promote_typejoin`](@ref), [`promote_rule`](@ref).
249
250 # Examples
251 ```jldoctest
252 julia> promote_type(Int64, Float64)
253 Float64
254
255 julia> promote_type(Int32, Int64)
256 Int64
257
258 julia> promote_type(Float32, BigInt)
259 BigFloat
260
261 julia> promote_type(Int16, Float16)
262 Float16
263
264 julia> promote_type(Int64, Float16)
265 Float16
266
267 julia> promote_type(Int8, UInt16)
268 UInt16
269 ```
270 """
