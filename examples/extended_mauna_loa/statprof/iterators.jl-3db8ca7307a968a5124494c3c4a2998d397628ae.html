180 """
181 pairs(IndexLinear(), A)

184
185 An iterator that accesses each element of the array `A`, returning
186 `i => x`, where `i` is the index for the element and `x = A[i]`.
187 Identical to `pairs(A)`, except that the style of index can be selected.
188 Also similar to `enumerate(A)`, except `i` will be a valid index

207 2 b
208 3 c

213 julia> S = view(A, 1:2, :);

290 "b"

294 julia> c = zip(a,b)

410 """
411 Iterators.filter(flt, itr)
412
413 Given a predicate function `flt` and an iterable object `itr`, return an
414 iterable object which upon iteration yields the elements `x` of `itr` that
415 satisfy `flt(x)`. The order of the original iterator is preserved.
416
417 This function is *lazy*; that is, it is guaranteed to return in ``Θ(1)`` time
418 and use ``Θ(1)`` additional space, and `flt` will not be called by an
419 invocation of `filter`. Calls to `flt` will be made when iterating over the
420 returned iterable object. These calls are not cached and repeated calls will be
421 made when reiterating.
422
423 See [`Base.filter`](@ref) for an eager implementation of filtering for arrays.
424
425 # Examples
426 ```jldoctest
427 julia> f = Iterators.filter(isodd, [1, 2, 3, 4, 5])
428 Base.Iterators.Filter{typeof(isodd), Vector{Int64}}(isodd, [1, 2, 3, 4, 5])
429
430 julia> foreach(println, f)
431 1
432 3
433 5
434
435 julia> [x for x in [1, 2, 3, 4, 5] if isodd(x)] # collects a generator over Iterators.filter
436 3-element Vector{Int64}:
437 1
438 3
439 5
440 ```
441 """
442 filter(flt, itr) = Filter(flt, itr)

453 end

1215
1216 There are several different ways to think about this iterator wrapper:
1217
1218 1. It provides a mutable wrapper around an iterator and
1219 its iteration state.
1220 2. It turns an iterator-like abstraction into a `Channel`-like
1221 abstraction.
1222 3. It's an iterator that mutates to become its own rest iterator
1223 whenever an item is produced.
1224
1225 `Stateful` provides the regular iterator interface. Like other mutable iterators
1226 (e.g. [`Channel`](@ref)), if iteration is stopped early (e.g. by a [`break`](@ref) in a [`for`](@ref) loop),
1227 iteration can be resumed from the same spot by continuing to iterate over the
1228 same iterator object (in contrast, an immutable iterator would restart from the
1229 beginning).

1243 'b': ASCII/Unicode U+0062 (category Ll: Letter, lowercase)
1244 'c': ASCII/Unicode U+0063 (category Ll: Letter, lowercase)
1245 'd': ASCII/Unicode U+0064 (category Ll: Letter, lowercase)
1246
1247 julia> collect(a)

1249 'e': ASCII/Unicode U+0065 (category Ll: Letter, lowercase)
1250 'f': ASCII/Unicode U+0066 (category Ll: Letter, lowercase)

1265 3
1266
1267 julia> sum(a) # Sum the remaining elements
1268 7
1269 ```
1270 """
1271 mutable struct Stateful{T, VS}
1272 itr::T
1273 # A bit awkward right now, but adapted to the new iteration protocol
1274 nextvalstate::Union{VS, Nothing}
1275 taken::Int
1276 @inline function Stateful{<:Any, Any}(itr::T) where {T}
1277 new{T, Any}(itr, iterate(itr), 0)
1278 end
1279 @inline function Stateful(itr::T) where {T}
1280 VS = approx_iter_type(T)
1281 return new{T, VS}(itr, iterate(itr)::VS, 0)
1282 end
1283 end
1284
1285 function reset!(s::Stateful{T,VS}, itr::T=s.itr) where {T,VS}
1286 s.itr = itr
1287 setfield!(s, :nextvalstate, iterate(itr))
1288 s.taken = 0
1289 s
1290 end
1291
1292 if Base === Core.Compiler
1293 approx_iter_type(a::Type) = Any
1294 else
1295 # Try to find an appropriate type for the (value, state tuple),
1296 # by doing a recursive unrolling of the iteration protocol up to
1297 # fixpoint.
1298 approx_iter_type(itrT::Type) = _approx_iter_type(itrT, Base._return_type(iterate, Tuple{itrT}))
1299 # Not actually called, just passed to return type to avoid
1300 # having to typesplit on Nothing
1301 function doiterate(itr, valstate::Union{Nothing, Tuple{Any, Any}})
1302 valstate === nothing && return nothing
1303 val, st = valstate
1304 return iterate(itr, st)
1305 end
1306 function _approx_iter_type(itrT::Type, vstate::Type)
1307 vstate <: Union{Nothing, Tuple{Any, Any}} || return Any
1308 vstate <: Union{} && return Union{}
1309 nextvstate = Base._return_type(doiterate, Tuple{itrT, vstate})
1310 return (nextvstate <: vstate ? vstate : Any)
1311 end
1312 end
1313
1314 convert(::Type{Stateful}, itr) = Stateful(itr)
1315
1316 @inline isdone(s::Stateful, st=nothing) = s.nextvalstate === nothing
1317
1318 @inline function popfirst!(s::Stateful)
1319 vs = s.nextvalstate
1320 if vs === nothing
1321 throw(EOFError())
1322 else
1323 val, state = vs
1324 Core.setfield!(s, :nextvalstate, iterate(s.itr, state))
1325 s.taken += 1
1326 return val
1327 end
1328 end
1329
1330 @inline function peek(s::Stateful, sentinel=nothing)
1331 ns = s.nextvalstate
1332 return ns !== nothing ? ns[1] : sentinel
1333 end
1334 @inline iterate(s::Stateful, state=nothing) = s.nextvalstate === nothing ? nothing : (popfirst!(s), nothing)
1335 IteratorSize(::Type{Stateful{T,VS}}) where {T,VS} = IteratorSize(T) isa HasShape ? HasLength() : IteratorSize(T)
1336 eltype(::Type{Stateful{T, VS}} where VS) where {T} = eltype(T)
1337 IteratorEltype(::Type{Stateful{T,VS}}) where {T,VS} = IteratorEltype(T)
1338 length(s::Stateful) = length(s.itr) - s.taken
1339

1341 only(x)
1342
1343 Returns the one and only element of collection `x`, and throws an `ArgumentError` if the
1344 collection has zero or multiple elements.
1345
1346 See also [`first`](@ref), [`last`](@ref).
1347
1348 !!! compat "Julia 1.4"
1349 This method requires at least Julia 1.4.
1350 """
1351 @propagate_inbounds function only(x)
1352 i = iterate(x)
1353 @boundscheck if i === nothing
1354 throw(ArgumentError("Collection is empty, must contain exactly 1 element"))
1355 end
1356 (ret, state) = i::NTuple{2,Any}
1357 @boundscheck if iterate(x, state) !== nothing
1358 throw(ArgumentError("Collection has multiple elements, must contain exactly 1 element"))
1359 end
1360 return ret
1361 end
1362
1363 # Collections of known size
1364 only(x::Ref) = x[]
1365 only(x::Number) = x
1366 only(x::Char) = x
1367 only(x::Tuple{Any}) = x[1]
1368 only(x::Tuple) = throw(
1369 ArgumentError("Tuple contains $(length(x)) elements, must contain exactly 1 element")
1370 )
1371 only(a::AbstractArray{<:Any, 0}) = @inbounds return a[]
1372 only(x::NamedTuple{<:Any, <:Tuple{Any}}) = first(x)
1373 only(x::NamedTuple) = throw(
1374 ArgumentError("NamedTuple contains $(length(x)) elements, must contain exactly 1 element")
1375 )
