236 function rrule(::typeof(/), A::AbstractVecOrMat{<:Real}, B::AbstractVecOrMat{<:Real})
237 Aᵀ, dA_pb = rrule(adjoint, A)
238 Bᵀ, dB_pb = rrule(adjoint, B)
239 Cᵀ, dS_pb = rrule(\, Bᵀ, Aᵀ)
240 C, dC_pb = rrule(adjoint, Cᵀ)
241 function slash_pullback(Ȳ)
242 # Optimization note: dAᵀ, dBᵀ, dC are calculated no matter which partial you want
243 _, dC = dC_pb(Ȳ)
244 _, dBᵀ, dAᵀ = dS_pb(unthunk(dC))
245
246 ∂A = last(dA_pb(unthunk(dAᵀ)))
247 ∂B = last(dB_pb(unthunk(dBᵀ)))
248
249 (NoTangent(), ∂A, ∂B)
250 end
251 return C, slash_pullback
252 end

258 function rrule(::typeof(\), A::AbstractVecOrMat{<:Real}, B::AbstractVecOrMat{<:Real})
259 project_A = ProjectTo(A)
260 project_B = ProjectTo(B)
261
262 Y = A \ B
263 function backslash_pullback(ȳ)
264 Ȳ = unthunk(ȳ)
265 ∂A = @thunk begin
266 B̄ = A' \ Ȳ
267 Ā = -B̄ * Y'
268 Ā = add!!(Ā, (B - A * Y) * B̄' / A')
269 Ā = add!!(Ā, A' \ Y * (Ȳ' - B̄'A))
270 project_A(Ā)
271 end
272 ∂B = @thunk project_B(A' \ Ȳ)
273 return NoTangent(), ∂A, ∂B
274 end
275 return Y, backslash_pullback
