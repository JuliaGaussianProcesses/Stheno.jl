434 function getindex(dtfd::DataTypeFieldDesc, i::Int)
435 layout_ptr = convert(Ptr{DataTypeLayout}, dtfd.dt.layout)
436 fd_ptr = layout_ptr + sizeof(DataTypeLayout)
437 layout = unsafe_load(layout_ptr)
438 fielddesc_type = (layout.flags >> 1) & 3
439 nfields = layout.nfields
440 @boundscheck ((1 <= i <= nfields) || throw(BoundsError(dtfd, i)))
441 if fielddesc_type == 0
442 return FieldDesc(unsafe_load(Ptr{FieldDescStorage{UInt8}}(fd_ptr), i))
443 elseif fielddesc_type == 1
444 return FieldDesc(unsafe_load(Ptr{FieldDescStorage{UInt16}}(fd_ptr), i))
445 elseif fielddesc_type == 2
446 return FieldDesc(unsafe_load(Ptr{FieldDescStorage{UInt32}}(fd_ptr), i))
447 else
448 # fielddesc_type == 3
449 return FieldDesc(true, true, 0, 0)
450 end

1087 """
1088 may_invoke_generator(method, atypes, sparams)
1089
1090 Computes whether or not we may invoke the generator for the given `method` on
1091 the given atypes and sparams. For correctness, all generated function are
1092 required to return monotonic answers. However, since we don't expect users to
1093 be able to successfully implement this criterion, we only call generated
1094 functions on concrete types. The one exception to this is that we allow calling
1095 generators with abstract types if the generator does not use said abstract type
1096 (and thus cannot incorrectly use it to break monotonicity). This function
1097 computes whether we are in either of these cases.
1098
1099 Unlike normal functions, the compilation heuristics still can't generate good dispatch
1100 in some cases, but this may still allow inference not to fall over in some limited cases.
1101 """
1102 function may_invoke_generator(method::MethodInstance)
1103 return may_invoke_generator(method.def::Method, method.specTypes, method.sparam_vals)
1104 end
1105 function may_invoke_generator(method::Method, @nospecialize(atypes), sparams::SimpleVector)
1106 # If we have complete information, we may always call the generator
1107 isdispatchtuple(atypes) && return true
1108
1109 # We don't have complete information, but it is possible that the generator
1110 # syntactically doesn't make use of the information we don't have. Check
1111 # for that.
1112
1113 # For now, only handle the (common, generated by the frontend case) that the
1114 # generator only has one method
1115 generator = method.generator
1116 isa(generator, Core.GeneratedFunctionStub) || return false
1117 gen_mthds = methods(generator.gen)::MethodList
1118 length(gen_mthds) == 1 || return false
1119
1120 generator_method = first(gen_mthds)
1121 nsparams = length(sparams)
1122 isdefined(generator_method, :source) || return false
1123 code = generator_method.source
1124 nslots = ccall(:jl_ir_nslots, Int, (Any,), code)
1125 at = unwrap_unionall(atypes)::DataType
1126 (nslots >= 1 + length(sparams) + length(at.parameters)) || return false
1127
1128 for i = 1:nsparams
1129 if isa(sparams[i], TypeVar)
1130 if (ast_slotflag(code, 1 + i) & SLOT_USED) != 0
1131 return false
1132 end
1133 end
1134 end
1135 for i = 1:length(at.parameters)
1136 if !isdispatchelem(at.parameters[i])
1137 if (ast_slotflag(code, 1 + i + nsparams) & SLOT_USED) != 0
1138 return false
1139 end
1140 end
1141 end
1142 return true
1143 end

1448 fbody = bodyfunction(basemethod::Method)
1449
1450 Find the keyword "body function" (the function that contains the body of the method
1451 as written, called after all missing keyword-arguments have been assigned default values).
1452 `basemethod` is the method you obtain via [`which`](@ref) or [`methods`](@ref).
1453 """
1454 function bodyfunction(basemethod::Method)

1456 isa(arg, Symbol) && return arg
1457 isa(arg, GlobalRef) && return arg.name
1458 return nothing
1459 end
1460
1461 fmod = basemethod.module
1462 # The lowered code for `basemethod` should look like
1463 # %1 = mkw(kwvalues..., #self#, args...)
1464 # return %1
1465 # where `mkw` is the name of the "active" keyword body-function.
1466 ast = Base.uncompressed_ast(basemethod)
1467 f = nothing
1468 if isa(ast, Core.CodeInfo) && length(ast.code) >= 2
1469 callexpr = ast.code[end-1]
1470 if isa(callexpr, Expr) && callexpr.head == :call
1471 fsym = callexpr.args[1]
1472 if isa(fsym, Symbol)
1473 f = getfield(fmod, fsym)
1474 elseif isa(fsym, GlobalRef)
1475 newsym = nothing
1476 if fsym.mod === Core && fsym.name === :_apply
1477 newsym = getsym(callexpr.args[2])
1478 elseif fsym.mod === Core && fsym.name === :_apply_iterate
1479 newsym = getsym(callexpr.args[3])
1480 end
1481 if isa(newsym, Symbol)
1482 f = getfield(basemethod.module, newsym)::Function
1483 else
1484 f = getfield(fsym.mod, fsym.name)::Function
1485 end
1486 end
1487 end
1488 end
1489 return f
1490 end

1492 """
1493 Base.isambiguous(m1, m2; ambiguous_bottom=false) -> Bool
1494
1495 Determine whether two methods `m1` and `m2` may be ambiguous for some call
1496 signature. This test is performed in the context of other methods of the same
1497 function; in isolation, `m1` and `m2` might be ambiguous, but if a third method
1498 resolving the ambiguity has been defined, this returns `false`.
1499 Alternatively, in isolation `m1` and `m2` might be ordered, but if a third
1500 method cannot be sorted with them, they may cause an ambiguity together.
1501
1502 For parametric types, the `ambiguous_bottom` keyword argument controls whether
1503 `Union{}` counts as an ambiguous intersection of type parameters – when `true`,
1504 it is considered ambiguous, when `false` it is not.
1505
1506 # Examples
1507 ```jldoctest
1508 julia> foo(x::Complex{<:Integer}) = 1
1509 foo (generic function with 1 method)
1510
1511 julia> foo(x::Complex{<:Rational}) = 2
1512 foo (generic function with 2 methods)
1513
1514 julia> m1, m2 = collect(methods(foo));
1515
1516 julia> typeintersect(m1.sig, m2.sig)
1517 Tuple{typeof(foo), Complex{Union{}}}
1518
1519 julia> Base.isambiguous(m1, m2, ambiguous_bottom=true)
1520 true
1521
1522 julia> Base.isambiguous(m1, m2, ambiguous_bottom=false)
1523 false
1524 ```
1525 """
1526 function isambiguous(m1::Method, m2::Method; ambiguous_bottom::Bool=false)
1527 m1 === m2 && return false
1528 ti = typeintersect(m1.sig, m2.sig)
1529 ti === Bottom && return false
1530 function inner(ti)
1531 ti === Bottom && return false
1532 if !ambiguous_bottom
1533 has_bottom_parameter(ti) && return false
1534 end
1535 min = UInt[typemin(UInt)]
1536 max = UInt[typemax(UInt)]
1537 has_ambig = Int32[0]
1538 ms = _methods_by_ftype(ti, nothing, -1, typemax(UInt), true, min, max, has_ambig)::Vector
1539 has_ambig[] == 0 && return false
1540 if !ambiguous_bottom
1541 filter!(ms) do m::Core.MethodMatch
1542 return !has_bottom_parameter(m.spec_types)
1543 end
1544 end
1545 # if ml-matches reported the existence of an ambiguity over their
1546 # intersection, see if both m1 and m2 may be involved in it
1547 have_m1 = have_m2 = false
1548 for match in ms
1549 match = match::Core.MethodMatch
1550 m = match.method
1551 m === m1 && (have_m1 = true)
1552 m === m2 && (have_m2 = true)
1553 end
1554 if !have_m1 || !have_m2
1555 # ml-matches did not need both methods to expose the reported ambiguity
1556 return false
1557 end
1558 if !ambiguous_bottom
1559 # since we're intentionally ignoring certain ambiguities (via the
1560 # filter call above), see if we can now declare the intersection fully
1561 # covered even though it is partially ambiguous over Union{} as a type
1562 # parameter somewhere
1563 minmax = nothing
1564 for match in ms
1565 m = match.method
1566 match.fully_covers || continue
1567 if minmax === nothing || morespecific(m.sig, minmax.sig)
1568 minmax = m
1569 end
1570 end
1571 if minmax === nothing
1572 return true
1573 end
1574 for match in ms
1575 m = match.method
1576 m === minmax && continue
1577 if match.fully_covers
1578 if !morespecific(minmax.sig, m.sig)
1579 return true
1580 end
1581 else
1582 if morespecific(m.sig, minmax.sig)
1583 return true
1584 end
1585 end
1586 end
1587 return false
1588 end
1589 return true
1590 end
1591 if !(ti <: m1.sig && ti <: m2.sig)
1592 # When type-intersection fails, it's often also not commutative. Thus
1593 # checking the reverse may allow detecting ambiguity solutions
1594 # correctly in more cases (and faster).
1595 ti2 = typeintersect(m2.sig, m1.sig)
1596 if ti2 <: m1.sig && ti2 <: m2.sig
1597 ti = ti2
1598 elseif ti != ti2
1599 # TODO: this would be the correct way to handle this case, but
1600 # people complained so we don't do it
1601 # inner(ti2) || return false
1602 return false # report that the type system failed to decide if it was ambiguous by saying they definitely aren't
1603 else
1604 return false # report that the type system failed to decide if it was ambiguous by saying they definitely aren't
1605 end
1606 end
1607 inner(ti) || return false
1608 # otherwise type-intersection reported an ambiguity we couldn't solve
1609 return true
1610 end
