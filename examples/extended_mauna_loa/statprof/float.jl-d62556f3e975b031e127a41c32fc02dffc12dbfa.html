167 function Float64(x::UInt128)
168 x == 0 && return 0.0
169 n = 128-leading_zeros(x) # ndigits0z(x,2)
170 if n <= 53
171 y = ((x % UInt64) << (53-n)) & 0x000f_ffff_ffff_ffff
172 else
173 y = ((x >> (n-54)) % UInt64) & 0x001f_ffff_ffff_ffff # keep 1 extra bit
174 y = (y+1)>>1 # round, ties up (extra leading bit in case of next exponent)
175 y &= ~UInt64(trailing_zeros(x) == (n-54)) # fix last bit to round to even
176 end
177 d = ((n+1022) % UInt64) << 52

179 end

466 # Exact Float (Tf) vs Integer (Ti) comparisons
467 # Assumes:
468 # - typemax(Ti) == 2^n-1
469 # - typemax(Ti) can't be exactly represented by Tf:
470 # => Tf(typemax(Ti)) == 2^n or Inf
471 # - typemin(Ti) can be exactly represented by Tf
472 #
473 # 1. convert y::Ti to float fy::Tf
474 # 2. perform Tf comparison x vs fy
475 # 3. if x == fy, check if (1) resulted in rounding:
476 # a. convert fy back to Ti and compare with original y
477 # b. unsafe_convert undefined behaviour if fy == Tf(typemax(Ti))
478 # (but consequently x == fy > y)

483 fy = ($Tf)(y)
484 (x == fy) & (fy != $(Tf(typemax(Ti)))) & (y == unsafe_trunc($Ti,fy))
485 end
486 ==(y::$Ti, x::$Tf) = x==y

562 return hash(xu, h)
563 end
564 elseif isnan(x)
565 return hx_NaN ⊻ h # NaN does not have a stable bit pattern
566 end
567 return hash_uint64(bitcast(UInt64, x)) - 3h
568 end
569 end

572 hash(x::Float16, h::UInt) = hash(Float64(x), h)

574 ## generic hashing for rational values ##
575
576 function hash(x::Real, h::UInt)
577 # decompose x as num*2^pow/den
578 num, pow, den = decompose(x)
579
580 # handle special values
581 num == 0 && den == 0 && return hash(NaN, h)
582 num == 0 && return hash(ifelse(den > 0, 0.0, -0.0), h)
583 den == 0 && return hash(ifelse(num > 0, Inf, -Inf), h)
584
585 # normalize decomposition
586 if den < 0
587 num = -num
588 den = -den
589 end
590 z = trailing_zeros(num)
591 if z != 0
592 num >>= z
593 pow += z
594 end
595 z = trailing_zeros(den)
596 if z != 0
597 den >>= z
598 pow -= z
599 end
600
601 # handle values representable as Int64, UInt64, Float64
602 if den == 1
603 left = ndigits0z(num,2) + pow
604 right = trailing_zeros(num) + pow
605 if -1074 <= right
606 if 0 <= right && left <= 64
607 left <= 63 && return hash(Int64(num) << Int(pow), h)
608 signbit(num) == signbit(den) && return hash(UInt64(num) << Int(pow), h)
609 end # typemin(Int64) handled by Float64 case
610 left <= 1024 && left - right <= 53 && return hash(ldexp(Float64(num),pow), h)
611 end
612 end
613
614 # handle generic rational values
615 h = hash_integer(den, h)
616 h = hash_integer(pow, h)
617 h = hash_integer(num, h)
618 return h
619 end
620
621 #=
622 `decompose(x)`: non-canonical decomposition of rational values as `num*2^pow/den`.
623
624 The decompose function is the point where rational-valued numeric types that support
625 hashing hook into the hashing protocol. `decompose(x)` should return three integer
626 values `num, pow, den`, such that the value of `x` is mathematically equal to
627
628 num*2^pow/den
629
630 The decomposition need not be canonical in the sense that it just needs to be *some*
631 way to express `x` in this form, not any particular way – with the restriction that
632 `num` and `den` may not share any odd common factors. They may, however, have powers
633 of two in common – the generic hashing code will normalize those as necessary.
634
635 Special values:
636
637 - `x` is zero: `num` should be zero and `den` should have the same sign as `x`
638 - `x` is infinite: `den` should be zero and `num` should have the same sign as `x`
639 - `x` is not a number: `num` and `den` should both be zero
640 =#
641
642 decompose(x::Integer) = x, 0, 1
643
644 function decompose(x::Float16)::NTuple{3,Int}
645 isnan(x) && return 0, 0, 0
646 isinf(x) && return ifelse(x < 0, -1, 1), 0, 0
647 n = reinterpret(UInt16, x)
648 s = (n & 0x03ff) % Int16
649 e = ((n & 0x7c00) >> 10) % Int
650 s |= Int16(e != 0) << 10
651 d = ifelse(signbit(x), -1, 1)
652 s, e - 25 + (e == 0), d
653 end
654
655 function decompose(x::Float32)::NTuple{3,Int}
656 isnan(x) && return 0, 0, 0
657 isinf(x) && return ifelse(x < 0, -1, 1), 0, 0
658 n = reinterpret(UInt32, x)
659 s = (n & 0x007fffff) % Int32
660 e = ((n & 0x7f800000) >> 23) % Int
661 s |= Int32(e != 0) << 23
662 d = ifelse(signbit(x), -1, 1)
663 s, e - 150 + (e == 0), d
664 end
665
666 function decompose(x::Float64)::Tuple{Int64, Int, Int}
667 isnan(x) && return 0, 0, 0
668 isinf(x) && return ifelse(x < 0, -1, 1), 0, 0
669 n = reinterpret(UInt64, x)
670 s = (n & 0x000fffffffffffff) % Int64
671 e = ((n & 0x7ff0000000000000) >> 52) % Int
672 s |= Int64(e != 0) << 52
673 d = ifelse(signbit(x), -1, 1)
674 s, e - 1075 + (e == 0), d
675 end

704 nextfloat(x::AbstractFloat, n::Integer)
705
706 The result of `n` iterative applications of `nextfloat` to `x` if `n >= 0`, or `-n`
707 applications of [`prevfloat`](@ref) if `n < 0`.
708 """
709 function nextfloat(f::IEEEFloat, d::Integer)
710 F = typeof(f)
711 fumax = reinterpret(Unsigned, F(Inf))
712 U = typeof(fumax)
713
714 isnan(f) && return f
715 fi = reinterpret(Signed, f)
716 fneg = fi < 0
717 fu = unsigned(fi & typemax(fi))
718
719 dneg = d < 0
720 da = uabs(d)
721 if da > typemax(U)
722 fneg = dneg
723 fu = fumax
724 else
725 du = da % U
726 if fneg ⊻ dneg
727 if du > fu
728 fu = min(fumax, du - fu)
729 fneg = !fneg
730 else
731 fu = fu - du
732 end
733 else
734 if fumax - fu < du
735 fu = fumax
736 else
737 fu = fu + du
738 end
739 end
740 end
741 if fneg
742 fu |= sign_mask(F)
743 end
744 reinterpret(F, fu)
745 end

931 """
932 eps(x::AbstractFloat)
933
934 Return the *unit in last place* (ulp) of `x`. This is the distance between consecutive
935 representable floating point values at `x`. In most cases, if the distance on either side
936 of `x` is different, then the larger of the two is taken, that is
937
938 eps(x) == max(x-prevfloat(x), nextfloat(x)-x)
939
940 The exceptions to this rule are the smallest and largest finite values
941 (e.g. `nextfloat(-Inf)` and `prevfloat(Inf)` for [`Float64`](@ref)), which round to the
942 smaller of the values.
943
944 The rationale for this behavior is that `eps` bounds the floating point rounding
945 error. Under the default `RoundNearest` rounding mode, if ``y`` is a real number and ``x``
946 is the nearest floating point number to ``y``, then
947
948 ```math
949 |y-x| \\leq \\operatorname{eps}(x)/2.
950 ```
951
952 See also: [`nextfloat`](@ref), [`issubnormal`](@ref), [`floatmax`](@ref).
953
954 # Examples
955 ```jldoctest
956 julia> eps(1.0)
957 2.220446049250313e-16
958
959 julia> eps(prevfloat(2.0))
960 2.220446049250313e-16
961
962 julia> eps(2.0)
963 4.440892098500626e-16
964
965 julia> x = prevfloat(Inf) # largest finite Float64
966 1.7976931348623157e308
967
968 julia> x + eps(x)/2 # rounds up
969 Inf
970
971 julia> x + prevfloat(eps(x)/2) # rounds down
972 1.7976931348623157e308
973 ```
974 """
