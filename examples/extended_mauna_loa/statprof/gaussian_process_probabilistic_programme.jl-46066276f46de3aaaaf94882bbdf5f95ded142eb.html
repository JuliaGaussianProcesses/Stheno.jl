113 """
114 Base.split(x::BlockData, Y::AbstractVecOrMat)
115
116 Convenience functionality to make working with the output of operations on GPPPs easier.
117 Splits up the rows of `Y` according to the sizes of the data in `x`.
118
119 ```jldoctest
120 julia> Y = vcat(randn(5, 3), randn(4, 3));
121
122 julia> x = BlockData(randn(5), randn(4));
123
124 julia> Y1, Y2 = split(x, Y);
125
126 julia> Y1 == Y[1:5, :]
127 true
128
129 julia> Y2 == Y[6:end, :]
130 true
131 ```
132
133 Works with any `BlockData`, so blocks can e.g. be `GPPPInput`s. This is particularly helpful
134 for working with the output from `rand` and `marginals` from a `GPPP` indexed at
135 `BlockData`. For example
136 ```julia
137 f = @gppp let
138 f1 = GP(SEKernel())
139 f2 = GP(Matern52Kernel())
140 f3 = f1 + f2
141 end
142
143 x = BlockData(GPPPInput(:f2, randn(3)), GPPPInput(:f3, randn(4)))
144 y = rand(f(x))
145 y2, y3 = split(x, y)
146 ```
147
148 Functionality also works with any `AbstractVector`.
149 """

168 """
169 @gppp(model_expression)
170
171 Construct a `GaussianProcessProbabilisticProgramme` (`GPPP`) from a code snippet.
172
173 ```jldoctest
174 f = @gppp let
175 f1 = GP(SEKernel())
176 f2 = GP(Matern52Kernel())
177 f3 = f1 + f2
178 end
179
180 x_local = randn(5)
181
182 x = BlockData(GPPPInput(:f1, x_local), GPPPInput(:f2, x_local), GPPPInput(:f3, x_local))
183
184 y = rand(f(x, 1e-12))
185
186 f1, f2, f3 = split(x, y)
187
188 isapprox(f1 + f2, f3; rtol=1e-4)
189
190 # output
191
192 true
193 ```
194 """
195 macro gppp(let_block::Expr)
196 # Consult the `Internals` section of the docs for info about what this macro is doing.
197
198 # Ensure that we're dealing with a let block.
199 let_block.head == :let || throw(error("gppp needs a let block."))
200
201 # Pull out model definition from the block.
202 model_expr = let_block.args[2]
203 model_expr.head == :block || throw(error("Expected a block."))
204
205 # Pull out the names of all of the variables.
206 lines_declaring_variables = filter(x -> x isa Expr && x.head == :(=), model_expr.args)
207 variable_names = map(x -> x.args[1], lines_declaring_variables)
208
209 # Construct expression which specifies mappings between symbolic names and GPs.
210 # The resulting expression is of the form (f1 = f1, f2 = f2).
211 var_mapping = Expr(
212 :tuple,
213 map(variable_names) do var_name
214 Expr(:(=), var_name, var_name)
215 end...,
216 )
217
218 gpc_sym = gensym("gpc")
219
220 # Construct an expression which wraps the GPs, and returns a GPPP.
221 wrapped_model = Expr(:block,
222 :($gpc_sym = Stheno.GPC()),
223 postwalk(
224 x->@capture(x, GP(xs__)) ? :(Stheno.wrap(GP($(xs...)), $gpc_sym)) : x,
225 model_expr,
226 ).args...,
227 :(Stheno.GPPP($var_mapping, $gpc_sym)),
228 )
229 return esc(wrapped_model)
230 end
