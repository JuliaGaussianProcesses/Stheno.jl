200 function check_reducedims(R, A)
201 # Check whether R has compatible dimensions w.r.t. A for reduction
202 #
203 # It returns an integer value (useful for choosing implementation)
204 # - If it reduces only along leading dimensions, e.g. sum(A, dims=1) or sum(A, dims=(1,2)),
205 # it returns the length of the leading slice. For the two examples above,
206 # it will be size(A, 1) or size(A, 1) * size(A, 2).
207 # - Otherwise, e.g. sum(A, dims=2) or sum(A, dims=(1,3)), it returns 0.
208 #
209 ndims(R) <= ndims(A) || throw(DimensionMismatch("cannot reduce $(ndims(A))-dimensional array to $(ndims(R)) dimensions"))
210 lsiz = 1
211 had_nonreduc = false
212 for i = 1:ndims(A)
213 Ri, Ai = axes(R, i), axes(A, i)
214 sRi, sAi = length(Ri), length(Ai)
215 if sRi == 1
216 if sAi > 1
217 if had_nonreduc
218 lsiz = 0 # to reduce along i, but some previous dimensions were non-reducing
219 else
220 lsiz *= sAi # if lsiz was set to zero, it will stay to be zero
221 end
222 end
223 else
224 Ri == Ai || throw(DimensionMismatch("reduction on array with indices $(axes(A)) with output with indices $(axes(R))"))
225 had_nonreduc = true
226 end
227 end
228 return lsiz
229 end
