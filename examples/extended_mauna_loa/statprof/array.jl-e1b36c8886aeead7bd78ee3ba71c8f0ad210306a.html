247 function _unsafe_copyto!(dest, doffs, src, soffs, n)
248 destp = pointer(dest, doffs)
249 srcp = pointer(src, soffs)
250 @inbounds if destp < srcp || destp > srcp + n
251 for i = 1:n
252 if isassigned(src, soffs + i - 1)
253 dest[doffs + i - 1] = src[soffs + i - 1]
254 else
255 _unsetindex!(dest, doffs + i - 1)
256 end
257 end
258 else

263 _unsetindex!(dest, doffs + i - 1)

271 unsafe_copyto!(dest::Array, do, src::Array, so, N)
272
273 Copy `N` elements from a source array to a destination, starting at offset `so` in the
274 source and `do` in the destination (1-indexed).
275
276 The `unsafe` prefix on this function indicates that no validation is performed to ensure
277 that N is inbounds on either array. Incorrect usage may corrupt or segfault your program, in
278 the same manner as C.

280 function unsafe_copyto!(dest::Array{T}, doffs, src::Array{T}, soffs, n) where T
281 t1 = @_gc_preserve_begin dest
282 t2 = @_gc_preserve_begin src
283 destp = pointer(dest, doffs)
284 srcp = pointer(src, soffs)
285 if !allocatedinline(T)
286 ccall(:jl_array_ptr_copy, Cvoid, (Any, Ptr{Cvoid}, Any, Ptr{Cvoid}, Int),
287 dest, destp, src, srcp, n)
288 elseif isbitstype(T)
289 ccall(:memmove, Ptr{Cvoid}, (Ptr{Cvoid}, Ptr{Cvoid}, Csize_t),

1474 function _deleteat!(a::Vector, inds, dltd=Nowhere())
1475 n = length(a)
1476 y = iterate(inds)
1477 y === nothing && return a
1478 (p, s) = y
1479 checkbounds(a, p)
1480 push!(dltd, @inbounds a[p])
1481 q = p+1
1482 while true
1483 y = iterate(inds, s)
1484 y === nothing && break
1485 (i,s) = y
1486 if !(q <= i <= n)
1487 if i < q
1488 throw(ArgumentError("indices must be unique and sorted"))
1489 else
1490 throw(BoundsError())
1491 end
1492 end
1493 while q < i
1494 @inbounds a[p] = a[q]
1495 p += 1; q += 1
1496 end
1497 push!(dltd, @inbounds a[i])
1498 q = i+1
1499 end
1500 while q <= n
1501 @inbounds a[p] = a[q]
1502 p += 1; q += 1
1503 end
1504 _deleteend!(a, n-p+1)
1505 return a
1506 end

2372 # If two collections are already sorted, _findin can be computed with
2373 # a single traversal of the two collections. This is much faster than
2374 # using a hash table (although it has the same complexity).
2375 function _sortedfindin(v::Union{AbstractArray, Tuple}, w)
2376 viter, witer = keys(v), eachindex(w)
2377 out = eltype(viter)[]
2378 vy, wy = iterate(viter), iterate(witer)
2379 if vy === nothing || wy === nothing
2380 return out
2381 end
2382 viteri, i = vy
2383 witerj, j = wy
2384 @inbounds begin
2385 vi, wj = v[viteri], w[witerj]
2386 while true
2387 if isless(vi, wj)
2388 vy = iterate(viter, i)
2389 if vy === nothing
2390 break
2391 end
2392 viteri, i = vy
2393 vi = v[viteri]
2394 elseif isless(wj, vi)
2395 wy = iterate(witer, j)
2396 if wy === nothing
2397 break
2398 end
2399 witerj, j = wy
2400 wj = w[witerj]
2401 else
2402 push!(out, viteri)
2403 vy = iterate(viter, i)
2404 if vy === nothing
2405 break
2406 end
2407 # We only increment the v iterator because v can have
2408 # repeated matches to a single value in w
2409 viteri, i = vy
2410 vi = v[viteri]
2411 end
2412 end
2413 end
2414 return out
