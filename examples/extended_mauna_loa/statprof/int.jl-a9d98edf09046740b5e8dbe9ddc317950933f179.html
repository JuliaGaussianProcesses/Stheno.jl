677
678 Parse a string into a [`BigInt`](@ref) or [`BigFloat`](@ref),
679 and throw an `ArgumentError` if the string is not a valid number.
680 For integers `_` is allowed in the string as a separator.

691 macro big_str(s)
692 if '_' in s
693 # remove _ in s[2:end-1]
694 bf = IOBuffer(maxsize=lastindex(s))
695 print(bf, s[1])
696 for c in SubString(s, 2, lastindex(s)-1)
697 c != '_' && print(bf, c)

700 seekstart(bf)

705 n === nothing || return n
706 n = tryparse(BigFloat, s)
707 n === nothing || return n
708 end
709 message = "invalid number format $s for BigInt or BigFloat"
710 return :(throw(ArgumentError($message)))

818 function widemul(u::Int64, v::Int64)
819 local u0::UInt64, v0::UInt64, w0::UInt64
820 local u1::Int64, v1::Int64, w1::UInt64, w2::Int64, t::UInt64
821
822 u0 = u & 0xffffffff; u1 = u >> 32
823 v0 = v & 0xffffffff; v1 = v >> 32

825 t = reinterpret(UInt64, u1) * v0 + (w0 >>> 32)
826 w2 = reinterpret(Int64, t) >> 32

829 lo = w0 & 0xffffffff + (w1 << 32)

831 end

870 function _setbit(x::UInt128, i)
871 # faster version of `return x | (UInt128(1) << i)`
872 j = i >> 5
873 y = UInt128(one(UInt32) << (i & 0x1f))
874 if j == 0
875 return x | y
876 elseif j == 1
877 return x | (y << 32)
878 elseif j == 2
879 return x | (y << 64)
880 elseif j == 3
881 return x | (y << 96)
882 end
883 return x
884 end

886 function divrem(x::UInt128, y::UInt128)
887 iszero(y) && throw(DivideError())
888 if (x >> 64) % UInt64 == 0
889 if (y >> 64) % UInt64 == 0
890 # fast path: upper 64 bits are zero, so we can fallback to UInt64 division
891 q64, x64 = divrem(x % UInt64, y % UInt64)
892 return UInt128(q64), UInt128(x64)
893 else
894 # this implies y>x, so
895 return zero(UInt128), x
896 end
897 end
898 n = leading_zeros(y) - leading_zeros(x)
899 q = zero(UInt128)
900 ys = y << n
901 while n >= 0
902 # ys == y * 2^n
903 if ys <= x
904 x -= ys
905 q = _setbit(q, n)
906 if (x >> 64) % UInt64 == 0
907 # exit early, similar to above fast path
908 if (y >> 64) % UInt64 == 0
909 q64, x64 = divrem(x % UInt64, y % UInt64)
910 q |= q64
911 x = UInt128(x64)
912 end
913 return q, x
914 end
915 end
916 ys >>>= 1
917 n -= 1
918 end
919 return q, x
920 end

986 const _mask1_uint128 = (UInt128(0x5555555555555555) << 64) | UInt128(0x5555555555555555)
987 const _mask2_uint128 = (UInt128(0x3333333333333333) << 64) | UInt128(0x3333333333333333)

990 """
991 bitreverse(x)
992
993 Reverse the order of bits in integer `x`. `x` must have a fixed bit width,
994 e.g. be an `Int16` or `Int32`.
995
996 !!! compat "Julia 1.5"
997 This function requires Julia 1.5 or later.
998
999 # Examples
1000 ```jldoctest
1001 julia> bitreverse(0x8080808080808080)
1002 0x0101010101010101
1003
1004 julia> reverse(bitstring(0xa06e)) == bitstring(bitreverse(0xa06e))
1005 true
1006 ```
1007 """
1008 function bitreverse(x::BitInteger)
1009 # TODO: consider using llvm.bitreverse intrinsic
1010 z = unsigned(x)
1011 mask1 = _mask1_uint128 % typeof(z)
1012 mask2 = _mask2_uint128 % typeof(z)
1013 mask4 = _mask4_uint128 % typeof(z)
1014 z = ((z & mask1) << 1) | ((z >> 1) & mask1)
1015 z = ((z & mask2) << 2) | ((z >> 2) & mask2)
1016 z = ((z & mask4) << 4) | ((z >> 4) & mask4)
1017 return bswap(z) % typeof(x)
1018 end
