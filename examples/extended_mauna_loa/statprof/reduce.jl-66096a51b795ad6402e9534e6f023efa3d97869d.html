53 function _foldl_impl(op::OP, init, itr) where {OP}
54 # Unroll the while loop once; if init is known, the call to op may
55 # be evaluated at compile time
56 y = iterate(itr)
57 y === nothing && return init
58 v = op(init, y[1])
59 while true
60 y = iterate(itr, y[2])
61 y === nothing && break
62 v = op(v, y[1])
63 end
64 return v
65 end

124 """
125 _xfadjoint(op, itr) -> op′, itr′
126
127 Given a pair of reducing function `op` and an iterator `itr`, return a pair
128 `(op′, itr′)` of similar types. If the iterator `itr` is transformed by an
129 iterator transform `ixf` whose adjoint transducer `xf` is known, `op′ = xf(op)`
130 and `itr′ = ixf⁻¹(itr)` is returned. Otherwise, `op` and `itr` are returned
131 as-is. For example, transducer `rf -> MappingRF(f, rf)` is the adjoint of
132 iterator transform `itr -> Generator(f, itr)`.
133
134 Nested iterator transforms are converted recursively. That is to say,
135 given `op` and
136
137 itr = (ixf₁ ∘ ixf₂ ∘ ... ∘ ixfₙ)(itr′)
138
139 what is returned is `itr′` and

141 op′ = (xfₙ ∘ ... ∘ xf₂ ∘ xf₁)(op)

262 """
263 mapreduce(f, op, itrs...; [init])
264
265 Apply function `f` to each element(s) in `itrs`, and then reduce the result using the binary
266 function `op`. If provided, `init` must be a neutral element for `op` that will be returned
267 for empty collections. It is unspecified whether `init` is used for non-empty collections.
268 In general, it will be necessary to provide `init` to work with empty collections.
269
270 [`mapreduce`](@ref) is functionally equivalent to calling
271 `reduce(op, map(f, itr); init=init)`, but will in general execute faster since no
272 intermediate collection needs to be created. See documentation for [`reduce`](@ref) and
273 [`map`](@ref).
274
275 !!! compat "Julia 1.2"
276 `mapreduce` with multiple iterators requires Julia 1.2 or later.
277
278 # Examples
279 ```jldoctest
280 julia> mapreduce(x->x^2, +, [1:3;]) # == 1 + 4 + 9
281 14
282 ```
283
284 The associativity of the reduction is implementation-dependent. Additionally, some
285 implementations may reuse the return value of `f` for elements that appear multiple times in
286 `itr`. Use [`mapfoldl`](@ref) or [`mapfoldr`](@ref) instead for
287 guaranteed left or right associativity and invocation of `f` for every value.

426 reduce(op, itr; [init])
427
428 Reduce the given collection `itr` with the given binary operator `op`. If provided, the
429 initial value `init` must be a neutral element for `op` that will be returned for empty
430 collections. It is unspecified whether `init` is used for non-empty collections.
431
432 For empty collections, providing `init` will be necessary, except for some special cases
433 (e.g. when `op` is one of `+`, `*`, `max`, `min`, `&`, `|`) when Julia can determine the
434 neutral element of `op`.
435
436 Reductions for certain commonly-used operators may have special implementations, and
437 should be used instead: `maximum(itr)`, `minimum(itr)`, `sum(itr)`, `prod(itr)`,
438 `any(itr)`, `all(itr)`.
439
440 The associativity of the reduction is implementation dependent. This means that you can't
441 use non-associative operations like `-` because it is undefined whether `reduce(-,[1,2,3])`
442 should be evaluated as `(1-2)-3` or `1-(2-3)`. Use [`foldl`](@ref) or
443 [`foldr`](@ref) instead for guaranteed left or right associativity.

447 elements are not reordered if you use an ordered collection.

605 isbadzero(::typeof(max), x::AbstractFloat) = (x == zero(x)) & signbit(x)

611 function mapreduce_impl(f, op::Union{typeof(max), typeof(min)},
612 A::AbstractArrayOrBroadcasted, first::Int, last::Int)
613 a1 = @inbounds A[first]
614 v1 = mapreduce_first(f, op, a1)
615 v2 = v3 = v4 = v1
616 chunk_len = 256
617 start = first + 1
618 simdstop = start + chunk_len - 4
619 while simdstop <= last - 3
620 # short circuit in case of NaN or missing
621 (v1 == v1) === true || return v1
622 (v2 == v2) === true || return v2
623 (v3 == v3) === true || return v3
624 (v4 == v4) === true || return v4
625 @inbounds for i in start:4:simdstop
626 v1 = _fast(op, v1, f(A[i+0]))
627 v2 = _fast(op, v2, f(A[i+1]))
628 v3 = _fast(op, v3, f(A[i+2]))
629 v4 = _fast(op, v4, f(A[i+3]))
630 end
631 checkbounds(A, simdstop+3)
632 start += chunk_len
633 simdstop += chunk_len
634 end
635 v = op(op(v1,v2),op(v3,v4))
636 for i in start:last
637 @inbounds ai = A[i]
638 v = op(v, f(ai))
639 end
640
641 # enforce correct order of 0.0 and -0.0
642 # e.g. maximum([0.0, -0.0]) === 0.0
643 # should hold
644 if isbadzero(op, v)
645 for i in first:last
646 x = @inbounds A[i]
647 isgoodzero(op,x) && return x
648 end
649 end
650 return v
651 end

1175 count([f=identity,] itr; init=0) -> Integer
1176
1177 Count the number of elements in `itr` for which the function `f` returns `true`.
1178 If `f` is omitted, count the number of `true` elements in `itr` (which
1179 should be a collection of boolean values). `init` optionally specifies the value
1180 to start counting from and therefore also determines the output type.
1181
1182 !!! compat "Julia 1.6"
1183 `init` keyword was added in Julia 1.6.
1184
1185 See also: [`any`](@ref), [`sum`](@ref).
1186
1187 # Examples
1188 ```jldoctest

1194

1196 0x07
1197 ```
1198 """
1199 count(itr; init=0) = count(identity, itr; init)
1200
1201 count(f, itr; init=0) = _simple_count(f, itr, init)
1202
1203 function _simple_count(pred, itr, init::T) where {T}
1204 n::T = init
1205 for x in itr
1206 n += pred(x)::Bool
1207 end
1208 return n
1209 end
1210
1211 function _simple_count(::typeof(identity), x::Array{Bool}, init::T=0) where {T}
1212 n::T = init
1213 chunks = length(x) ÷ sizeof(UInt)
1214 mask = 0x0101010101010101 % UInt
1215 GC.@preserve x begin
1216 ptr = Ptr{UInt}(pointer(x))
1217 for i in 1:chunks
1218 n = (n + count_ones(unsafe_load(ptr, i) & mask)) % T
1219 end
1220 end
1221 for i in sizeof(UInt)*chunks+1:length(x)
1222 n = (n + x[i]) % T
1223 end
1224 return n
