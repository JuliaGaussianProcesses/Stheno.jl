1 """
2 Tangent{P, T} <: AbstractTangent
3
4 This type represents the tangent for a `struct`/`NamedTuple`, or `Tuple`.
5 `P` is the the corresponding primal type that this is a tangent for.
6
7 `Tangent{P}` should have fields (technically properties), that match to a subset of the
8 fields of the primal type; and each should be a tangent type matching to the primal
9 type of that field.
10 Fields of the P that are not present in the Tangent are treated as `Zero`.
11
12 `T` is an implementation detail representing the backing data structure.
13 For Tuple it will be a Tuple, and for everything else it will be a `NamedTuple`.
14 It should not be passed in by user.
15
16 For `Tangent`s of `Tuple`s, `iterate` and `getindex` are overloaded to behave similarly
17 to for a tuple.
18 For `Tangent`s of `struct`s, `getproperty` is overloaded to allow for accessing values
19 via `tangent.fieldname`.
20 Any fields not explictly present in the `Tangent` are treated as being set to `ZeroTangent()`.
21 To make a `Tangent` have all the fields of the primal the [`canonicalize`](@ref)
22 function is provided.
23 """

185 """
186 canonicalize(tangent::Tangent{P}) -> Tangent{P}
187
188 Return the canonical `Tangent` for the primal type `P`.
189 The property names of the returned `Tangent` match the field names of the primal,
190 and all fields of `P` not present in the input `tangent` are explictly set to `ZeroTangent()`.
191 """
192 function canonicalize(tangent::Tangent{P,<:NamedTuple{L}}) where {P,L}
193 nil = _zeroed_backing(P)
194 combined = merge(nil, backing(tangent))
195 if length(combined) !== fieldcount(P)
196 throw(
197 ArgumentError(
198 "Tangent fields do not match primal fields.\n" *
199 "Tangent fields: $L. Primal ($P) fields: $(fieldnames(P))",
200 ),
201 )
202 end
203 return Tangent{P,typeof(combined)}(combined)
204 end

262 function elementwise_add(a::NamedTuple{an}, b::NamedTuple{bn}) where {an,bn}
263 # Rule of Tangent addition: any fields not present are implict hard Zeros
264
265 # Base on the `merge(:;NamedTuple, ::NamedTuple)` code from Base.
266 # https://github.com/JuliaLang/julia/blob/592748adb25301a45bd6edef3ac0a93eed069852/base/namedtuple.jl#L220-L231
267 if @generated
268 names = Base.merge_names(an, bn)
269
270 vals = map(names) do field
271 a_field = :(getproperty(a, $(QuoteNode(field))))
272 b_field = :(getproperty(b, $(QuoteNode(field))))
273 value_expr = if Base.sym_in(field, an)
274 if Base.sym_in(field, bn)
275 # in both
276 :($a_field + $b_field)
277 else
278 # only in `an`
279 a_field
280 end
281 else # must be in `b` only
282 b_field
283 end
284 Expr(:kw, field, value_expr)
285 end
286 return Expr(:tuple, Expr(:parameters, vals...))
287 else
288 names = Base.merge_names(an, bn)
289 vals = map(names) do field
290 value = if Base.sym_in(field, an)
291 a_field = getproperty(a, field)
292 if Base.sym_in(field, bn)
293 # in both
294 b_field = getproperty(b, field)
295 a_field + b_field
296 else
297 # only in `an`
298 a_field
299 end
300 else # must be in `b` only
301 getproperty(b, field)
302 end
303 field => value
304 end
305 return (; vals...)
